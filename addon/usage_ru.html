Подключите Style.php в файле main.php и используйте: 
  require_once("qrc://%%projectname%%/%%addonpath%%/Style.php");

  /* Инициализировать класс */
  $style = new Style("%child_dir%");

  or

  new Style("%child_dir%");

  /* Задаем тему оформления для приложения */
  $style->setTheme("%theme_name%");

  or

  Style::setTheme("%theme_name%");
  
  class MainWindow extends QWidget {

    public function __construct() {

      parent::__consctruct();

      /* Задаем стиль элементу */
      $this->styleSheet = Style::get("%filename%");

      or

      Style::set($this, "%filename%");

    }

    /* ... */
    
  }
  
  /* ... */
